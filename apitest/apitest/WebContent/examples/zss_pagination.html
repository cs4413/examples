zRSSFeed - Paging though RSS feeds using pagination

This examples shows paging through many feeds items using pagination links.

Script

$(document).ready(function () {
	$('#zslider').rssfeed('http://feeds.reuters.com/reuters/peopleNews',{
		header: false,
		titletag: 'div',
		limit: 16
	}, function(e) {
		$.zazar.slider({
			selector: '#zslider ul',
			speed: 200,
			pagination: '#pages',
			auto: false
		});
	});

});
