RSS demo using JQuery

Script

$(document).ready(function() {

	$("select").change(function(){
		var rssVal = $(this).val();
		var rssUrl = "../controllers/rssNewsController.php?q="+rssVal;
	    $.ajax({type: "GET",
	    	    url: rssUrl, 
	    	    dataType: 'text',
	    	    success: function(result){$("#rssOutput").html(result);},
	            error: function() {
	            	alert ('Failed to download feed');
	            }
	           });
	  });
});
